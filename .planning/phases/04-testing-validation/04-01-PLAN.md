### ERR-01: YAML parse warnings surface in RuntimeSummaryService.metadata.warnings

**Critical**: This test asserts the contract from D-08: YAML parse/schema errors must appear in `RuntimeSummaryService.metadata.warnings`, NOT only in `console.warn()`. The current implementation logs to `console.warn()` but has no mechanism to propagate those warnings into the RuntimeSummaryService. If the test fails, that reveals the missing wiring — which is the expected outcome for Phase 4 gap detection.

```typescript
  describe('ERR-01: YAML parse warnings in RuntimeSummaryService.metadata.warnings', () => {
    it('getSummary({funnels}) metadata.warnings contains YAML schema validation message', () => {
      // Write YAML that parses but fails schema validation (missing funnels array)
      const invalidYaml = `version: "1.0"
# funnels key is missing — triggers schema validation
noop: true
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), invalidYaml, 'utf-8');

      // Create a loader and load the invalid YAML
      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();

      // The funnels map should be empty (last-known-good preserved as empty)
      expect(funnels.size).toBe(0);

      // RuntimeSummaryService must surface the YAML warning in metadata.warnings
      // per D-08: "YAML parse/schema errors → RuntimeSummaryService.metadata.warnings"
      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });
      const hasYamlWarning = summary.metadata.warnings.some(
        (w) => w.includes('workflows.yaml') || w.includes('funnel') || w.includes('validation')
      );
      expect(hasYamlWarning).toBe(true);
    });

    it('console.warn is NOT the only output path for YAML parse errors', () => {
      // Write malformed YAML that throws a parse exception
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), 'INVALID YAML: [}', 'utf-8');

      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();

      // console.warn was called by the loader (verified manually), but the
      // real requirement is that the warning also appears in RuntimeSummaryService
      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });

      // D-08 contract: warnings must be in metadata.warnings, not just console
      const hasParseWarning = summary.metadata.warnings.some(
        (w) => w.includes('parse') || w.includes('workflows.yaml') || w.includes('Failed')
      );
      expect(hasParseWarning).toBe(true);
    });
  });
```

### ERR-02: Missing or malformed YAML triggers degraded state in RuntimeSummaryService

**Critical**: Success Criterion 1 requires "status output shows explicit 'degraded' state with warning in metadata". The test must call `getSummary()` and verify the returned summary object has degraded indicators in its top-level fields, not just that funnels is empty.

The degraded state signals in `RuntimeSummary` are:
- `gfi.dataQuality === 'partial'` (not 'authoritative')
- `metadata.warnings` contains a warning about missing/invalid workflows.yaml

```typescript
  describe('ERR-02: degraded state on missing or malformed YAML', () => {
    it('getSummary shows degraded state when file is missing', () => {
      // No workflows.yaml file at all — loader creates empty funnels
      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();
      expect(funnels.size).toBe(0);

      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });

      // Success Criterion 1: degraded state with warning in metadata
      // gfi.dataQuality must be 'partial' (not 'authoritative') when funnels are absent
      expect(summary.gfi.dataQuality).toBe('partial');

      // metadata.warnings must contain a warning about the missing YAML config
      const hasConfigWarning = summary.metadata.warnings.some(
        (w) => w.toLowerCase().includes('workflow') || w.toLowerCase().includes('funnel')
      );
      expect(hasConfigWarning).toBe(true);
    });

    it('loader preserves last known-good funnels when YAML becomes malformed', () => {
      // First: load a valid funnel
      const validYaml = `version: "1.0"
funnels:
  - workflowId: good
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), validYaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      expect(loader.getAllFunnels().get('good')).toBeDefined();

      // Replace with completely malformed YAML
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), 'INVALID YAML: [}', 'utf-8');
      loader.load();

      // Last known-good must be preserved
      const funnels = loader.getAllFunnels();
      expect(funnels.get('good')).toBeDefined();
      expect(funnels.size).toBe(1);

      // getSummary must show degraded state AND warning about malformed YAML
      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });
      expect(summary.gfi.dataQuality).toBe('partial');
      const hasMalformedWarning = summary.metadata.warnings.some(
        (w) => w.includes('parse') || w.includes('workflows.yaml') || w.includes('Failed')
      );
      expect(hasMalformedWarning).toBe(true);
    });
  });
```

### ERR-03: last-known-good preserved on invalid YAML replacement

```typescript
  describe('ERR-03: last-known-good preserved on invalid YAML', () => {
    it('preserves funnel definitions when YAML is replaced with invalid content', () => {
      const validYaml = `version: "1.0"
funnels:
  - workflowId: preserved
    stages:
      - name: s1
        eventType: e1
        eventCategory: blocked
        statsField: evolution.e1
  - workflowId: also-preserved
    stages:
      - name: s2
        eventType: e2
        eventCategory: created
        statsField: evolution.e2
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), validYaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);

      // Verify both funnels loaded
      expect(loader.getAllFunnels().has('preserved')).toBe(true);
      expect(loader.getAllFunnels().has('also-preserved')).toBe(true);

      // Corrupt the YAML file
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), 'truly invalid: :::', 'utf-8');
      loader.load();

      // Both must still be present — last known-good preserved
      expect(loader.getAllFunnels().has('preserved')).toBe(true);
      expect(loader.getAllFunnels().has('also-preserved')).toBe(true);
    });

    it('clears funnels only when file is genuinely missing, not on parse error', () => {
      // Missing file case — funnels empty
      const loaderMissing = new WorkflowFunnelLoader(tempDir);
      expect(loaderMissing.getAllFunnels().size).toBe(0);

      // Valid YAML then parse error
      const validYaml = `version: "1.0"
funnels:
  - workflowId: to-clear
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), validYaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      expect(loader.getAllFunnels().has('to-clear')).toBe(true);

      // Write YAML that parses but has wrong schema
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), 'version: "1.0"\nfunnels: null', 'utf-8');
      loader.load();
      // Must still have last known-good
      expect(loader.getAllFunnels().has('to-clear')).toBe(true);
    });
  });
```

### PLAT-01: Windows atomic-save rename/rewrite sequence

```typescript
  describe('PLAT-01: Windows atomic-save rename/rewrite sequence', () => {
    it('handles change event after file rewrite (Windows atomic-save pattern)', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: windows-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      expect(loader.getAllFunnels().get('windows-test')).toBeDefined();

      // Simulate Windows atomic-save: rewrite file with new content
      const updatedYaml = `version: "1.0"
funnels:
  - workflowId: windows-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
      - name: s2
        eventType: e2
        eventCategory: created
        statsField: evolution.e2
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), updatedYaml, 'utf-8');
      loader.load();  // Manual reload simulates what the watcher debounce would do

      const funnels = loader.getAllFunnels();
      expect(funnels.get('windows-test')?.length).toBe(2);
    });

    it('debounce timer is cleared on rapid successive changes', async () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: debounce-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      loader.watch();

      const v2 = `version: "1.0"
funnels:
  - workflowId: debounce-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
      - name: s2
        eventType: e2
        eventCategory: created
        statsField: evolution.e2
`;
      const v3 = `version: "1.0"
funnels:
  - workflowId: debounce-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
      - name: s2
        eventType: e2
        eventCategory: created
        statsField: evolution.e2
      - name: s3
        eventType: e3
        eventCategory: blocked
        statsField: evolution.e3
`;

      // Write v2 then v3 rapidly
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), v2, 'utf-8');
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), v3, 'utf-8');

      // Wait for debounce to settle (100ms debounce + margin)
      await new Promise((r) => setTimeout(r, 250));

      // After debounce settles, should have v3 content (3 stages)
      const stages = loader.getAllFunnels().get('debounce-test') ?? [];
      expect(stages.length).toBe(3);

      loader.dispose();
    });
  });
```
