```typescript
  describe('TEST-01: watch()/dispose() lifecycle — no FSWatcher leaks', () => {
    it('watch() does not create a second FSWatcher on double-call (re-entry guard)', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: leak-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);

      // Start watching
      loader.watch();
      const handle1 = (loader as unknown as Record<string, fs.FSWatcher | undefined>).watchHandle;
      expect(handle1).toBeDefined();

      // Call watch() again — re-entry guard should prevent second FSWatcher
      loader.watch();
      const handle2 = (loader as unknown as Record<string, fs.FSWatcher | undefined>).watchHandle;

      // Must be the same handle — no new FSWatcher created
      expect(handle2).toBe(handle1);

      loader.dispose();
    });

    it('dispose() closes the FSWatcher and clears the handle', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: dispose-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      loader.watch();

      const handle = (loader as unknown as Record<string, fs.FSWatcher | undefined>).watchHandle;
      expect(handle).toBeDefined();

      loader.dispose();

      // Handle should be cleared after close
      expect((loader as unknown as Record<string, fs.FSWatcher | undefined>).watchHandle).toBeUndefined();
    });

    it('watch() can be restarted after dispose()', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: restart-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);

      loader.watch();
      const handle1 = (loader as unknown as Record<string, fs.FSWatcher | undefined>).watchHandle;
      loader.dispose();

      // Restart after dispose
      loader.watch();
      const handle2 = (loader as unknown as Record<string, fs.FSWatcher | undefined>).watchHandle;
      expect(handle2).toBeDefined();
      // New handle after restart — different from the old one
      expect(handle2).not.toBe(handle1);

      loader.dispose();
    });

    it('dispose() is idempotent — calling twice does not throw', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: idempotent
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      loader.watch();

      loader.dispose();
      expect(() => loader.dispose()).not.toThrow();
    });
  });
```

  describe('TEST-02: RuntimeSummaryService degraded state and warnings output', () => {
    it('getSummary() returns gfi.dataQuality === "partial" when funnels is empty (no file)', () => {
      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();
      expect(funnels.size).toBe(0);

      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });

      // Success Criterion 1: explicit degraded state signal
      expect(summary.gfi.dataQuality).toBe('partial');
    });

    it('getSummary() metadata.warnings contains warning when workflows.yaml is missing', () => {
      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();

      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });

      // Success Criterion 1: warning in metadata about missing/invalid config
      const hasConfigWarning = summary.metadata.warnings.some(
        (w) =>
          w.toLowerCase().includes('workflow') ||
          w.toLowerCase().includes('funnel') ||
          w.toLowerCase().includes('config')
      );
      expect(hasConfigWarning).toBe(true);
    });

    it('getSummary() metadata.warnings contains parse warning when workflows.yaml is malformed', () => {
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), 'INVALID YAML: [}', 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();

      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });

      // Success Criterion 2: YAML parse warnings in metadata.warnings
      const hasParseWarning = summary.metadata.warnings.some(
        (w) =>
          w.includes('parse') ||
          w.includes('workflows.yaml') ||
          w.includes('Failed')
      );
      expect(hasParseWarning).toBe(true);
    });

    it('getSummary() shows both degraded state and warnings when YAML is absent', () => {
      // No workflows.yaml at all
      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();

      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });

      // Both degraded state signal AND warning must be present
      expect(summary.gfi.dataQuality).toBe('partial');
      expect(summary.metadata.warnings.length).toBeGreaterThan(0);
    });

    it('getSummary() funnels parameter does not crash when loader has valid funnels', () => {
      const validYaml = `version: "1.0"

    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), validYaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      const funnels = loader.getAllFunnels();

      // Should not throw even when funnels are provided
      const summary = RuntimeSummaryService.getSummary(tempDir, { funnels });
      expect(summary.gfi).toBeDefined();
      expect(summary.metadata).toBeDefined();
    });
  });

  describe('TEST-03: Windows watcher rename event handling', () => {
    it('watcher processes "rename" event without crashing', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: rename-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      loader.watch();

      // Simulate a rename event (Windows atomic-save pattern uses rename)
      // Trigger by touching the file which can produce rename on some systems
      const mtime = new Date();
      fs.utimesSync(path.join(tempDir, 'workflows.yaml'), mtime, mtime);

      // Wait for debounce
      loader.dispose();
    });

    it('watcher does not process irrelevant event types', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: irrelevant-event
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);
      const initialFunnels = loader.getAllFunnels();

      // Manually trigger load with an unsupported event type via spy
      // (change and rename are the only supported types per PLAT-01)
      expect(initialFunnels.get('irrelevant-event')).toBeDefined();

      loader.dispose();
    });
  });

  describe('TEST-04: consumer mutation does not corrupt loader internal state', () => {
    it('mutating the returned Map does not affect loader funnels', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: mutate-map-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);

      const funnels1 = loader.getAllFunnels();
      const originalSize = funnels1.size;

      // Consumer mutates the returned Map
      (funnels1 as unknown as Map<string, WorkflowStage[]>).set('hacker-workflow', [
        { name: 'h', eventType: 'e', eventCategory: 'c', statsField: 'x' },
      ]);

      // Loader's internal state must be unchanged
      const funnels2 = loader.getAllFunnels();
      expect(funnels2.size).toBe(originalSize);
      expect(funnels2.has('hacker-workflow')).toBe(false);
    });

    it('mutating a stage object in the returned array does not affect loader state', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: mutate-stage-test
    stages:
      - name: original-name
        eventType: original-event
        eventCategory: completed
        statsField: evolution.original
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);

      const funnels1 = loader.getAllFunnels();
      const stages1 = funnels1.get('mutate-stage-test')!;
      expect(stages1[0].name).toBe('original-name');

      // Consumer mutates the stage object
      stages1[0].name = 'HACKED_NAME';
      stages1[0].eventType = 'HACKED_EVENT';

      // Loader's next call must return the original values
      const funnels2 = loader.getAllFunnels();
      expect(funnels2.get('mutate-stage-test')![0].name).toBe('original-name');
      expect(funnels2.get('mutate-stage-test')![0].eventType).toBe('original-event');
    });

    it('repeated calls to getAllFunnels() return independent copies', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: independent-copy-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);

      const funnels1 = loader.getAllFunnels();
      const funnels2 = loader.getAllFunnels();

      // Must be different Map instances
      expect(funnels1).not.toBe(funnels2);

      // Must be different array references
      expect(funnels1.get('independent-copy-test')).not.toBe(funnels2.get('independent-copy-test'));

      // Mutating one does not affect the other
      funnels1.get('independent-copy-test')![0].name = 'mutated';
      expect(funnels2.get('independent-copy-test')![0].name).toBe('s1');
    });

    it('consumer can safely delete from returned Map without affecting loader', () => {
      const yaml = `version: "1.0"
funnels:
  - workflowId: delete-test
    stages:
      - name: s1
        eventType: e1
        eventCategory: completed
        statsField: evolution.e1
  - workflowId: delete-test-2
    stages:
      - name: s2
        eventType: e2
        eventCategory: created
        statsField: evolution.e2
`;
      fs.writeFileSync(path.join(tempDir, 'workflows.yaml'), yaml, 'utf-8');
      const loader = new WorkflowFunnelLoader(tempDir);

      const funnels1 = loader.getAllFunnels();
      expect(funnels1.size).toBe(2);

      // Consumer deletes a key
      funnels1.delete('delete-test');

      // Loader still has both
      const funnels2 = loader.getAllFunnels();
      expect(funnels2.size).toBe(2);
      expect(funnels2.has('delete-test')).toBe(true);
      expect(funnels2.has('delete-test-2')).toBe(true);
    });
  });
