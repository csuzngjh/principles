    export async function candidateShow(
      options: CandidateShowOptions & { ledgerAdapter?: LedgerAdapter },
    ): Promise<CandidateShowResult | null> {
      const candidate = await options.stateManager.getCandidate(options.candidateId);
      if (!candidate) {
        return null;
      }

      // NEW: Check for ledger entry
      let ledgerEntryId: string | null = null;
      if (options.ledgerAdapter) {
        const entry = options.ledgerAdapter.existsForCandidate(options.candidateId);
        if (entry) {
          ledgerEntryId = entry.id;
        }
      }

      return {
        candidateId: candidate.candidateId,
        artifactId: candidate.artifactId,
        taskId: candidate.taskId,
        title: candidate.title,
        description: candidate.description,
        confidence: candidate.confidence,
        sourceRunId: candidate.sourceRunId,
        status: candidate.status,
        createdAt: candidate.createdAt,
        ledgerEntryId,  // NEW
      };
    }

    export async function handleCandidateShow(opts: CandidateShowOptions): Promise<void> {
      const workspaceDir = resolveWorkspaceDir(opts.workspace);

          CREATE TABLE IF NOT EXISTS diagnostician_tasks (
            task_id TEXT PRIMARY KEY,
            status TEXT NOT NULL DEFAULT 'pending',
            attempt_count INTEGER NOT NULL DEFAULT 0,
            max_attempts INTEGER NOT NULL DEFAULT 3,
            last_error TEXT,
            created_at TEXT NOT NULL,
            updated_at TEXT NOT NULL

          CREATE TABLE IF NOT EXISTS diagnostician_runs (
            run_id TEXT PRIMARY KEY,
            task_id TEXT NOT NULL,
            status TEXT NOT NULL,
            attempt_number INTEGER NOT NULL,
            runner_kind TEXT NOT NULL,
            output_text TEXT,
            error_text TEXT,
            created_at TEXT NOT NULL,
            FOREIGN KEY (task_id) REFERENCES diagnostician_tasks(task_id)
          );

          CREATE TABLE IF NOT EXISTS diagnostician_commits (

      // Helper to run pd CLI command
      function runPdCommand(args: string[]): { stdout: string; stderr: string; status: number } {
        try {

          return { stdout, stderr: '', status: 0 };
        } catch (err: any) {
          return {
            stdout: err.stdout || '',
            stderr: err.stderr || '',
            status: err.status || 1,
          };
        }
      }
