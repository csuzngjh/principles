<objective>
Implement PiAiRuntimeAdapter that implements the PDRuntimeAdapter interface using @mariozechner/pi-ai for direct LLM completion. This adapter bypasses OpenClaw CLI and calls LLM directly, solving the m8-03 UAT blocking issue where main agent takes >300s.

Key deliverables:
1. Add `@mariozechner/pi-ai` dependency to principles-core
2. Add `"pi-ai"` to RuntimeKindSchema union
3. Create PiAiRuntimeAdapter class implementing all PDRuntimeAdapter methods
4. Export from adapter/index.ts and runtime-v2/index.ts
</objective>

## Step 3: Create PiAiRuntimeAdapter

Create `packages/principles-core/src/runtime-v2/adapter/pi-ai-runtime-adapter.ts`:

```typescript
import { getModel, complete } from '@mariozechner/pi-ai'
import type { AssistantMessage } from '@mariozechner/pi-ai'
import { Type } from '@sinclair/typebox'
import { PDRuntimeError } from '../error-categories.js'
import { DiagnosticianOutputV1Schema } from '../diagnostician-output.js'
import type { DiagnosticianOutputV1 } from '../diagnostician-output.js'
import type {
  PDRuntimeAdapter,
  StartRunInput,
  RunHandle,
  RunStatus,
  StructuredRunOutput,
  RuntimeCapabilities,
  RuntimeHealth,
} from '../runtime-protocol.js'

export interface PiAiRuntimeAdapterConfig {
  provider: string
  model: string
  apiKeyEnv: string
  maxRetries?: number
  timeoutMs?: number
  workspace: string
}

export class PiAiRuntimeAdapter implements PDRuntimeAdapter {
  private readonly config: PiAiRuntimeAdapterConfig
  private readonly runs = new Map<string, { status: RunStatus; output?: StructuredRunOutput }>()

  constructor(config: PiAiRuntimeAdapterConfig) {
    this.config = {
      maxRetries: 2,
      timeoutMs: 120_000,
      ...config,
    }
  }

  kind(): string {
    return 'pi-ai'
  }

  getCapabilities(): RuntimeCapabilities {
    return {
      streaming: false,
      cancellation: true,
      artifactDelivery: false,
    }
  }

  async healthCheck(): Promise<RuntimeHealth> {
    const apiKey = process.env[this.config.apiKeyEnv]
    if (!apiKey) {
      return {
        healthy: false,
        message: `API key not found in env: ${this.config.apiKeyEnv}`,
      }
    }
    try {
      const model = getModel(this.config.provider, this.config.model)
      return { healthy: true, message: 'ok' }
    } catch (err) {
      return {
        healthy: false,
        message: `getModel failed: ${err instanceof Error ? err.message : String(err)}`,
      }
    }
  }

  async startRun(input: StartRunInput): Promise<RunHandle> {
    const runId = `pi-ai-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`
    const apiKey = process.env[this.config.apiKeyEnv]
    if (!apiKey) {
      throw new PDRuntimeError(
        'runtime_unavailable',
        `API key not found in env: ${this.config.apiKeyEnv}`
      )
    }

    // Start async execution
    this.executeRun(runId, input, apiKey).catch(() => {})

    return {
      runId,
      status: 'running',
      provider: this.config.provider,
      model: this.config.model,
    }
  }

  private async executeRun(runId: string, input: StartRunInput, apiKey: string): Promise<void> {
    this.runs.set(runId, { status: 'running' })

    let lastError: Error | null = null
    const maxRetries = this.config.maxRetries ?? 2

    for (let attempt = 0; attempt <= maxRetries; attempt++) {
      try {
        const model = getModel(this.config.provider, this.config.model)
        const controller = new AbortController()
        const timeoutId = setTimeout(() => controller.abort(), this.config.timeoutMs!)

        const context = {
          systemPrompt: input.systemPrompt || '',
          messages: [{ role: 'user' as const, content: input.prompt }],
        }

        const response: AssistantMessage = await complete(model, context, {
          signal: controller.signal,
        })
        clearTimeout(timeoutId)

        // Extract text content
        const textContent = response.content
          .filter((c): c is { type: 'text'; text: string } => c.type === 'text')
          .map(c => c.text)
          .join('\n')

        // Parse and validate output
        let output: DiagnosticianOutputV1
        try {
          const parsed = JSON.parse(textContent)
          output = parsed as DiagnosticianOutputV1
          // Basic validation
          if (!output.valid || !output.diagnosisId || !output.summary) {
            throw new Error('Missing required fields in DiagnosticianOutputV1')
          }
        } catch (parseErr) {
          throw new PDRuntimeError(
            'output_invalid',
            `Failed to parse DiagnosticianOutputV1: ${parseErr instanceof Error ? parseErr.message : String(parseErr)}`
          )
        }

        this.runs.set(runId, {
          status: 'succeeded',
          output: {
            runId,
            status: 'succeeded',
            output: JSON.stringify(output),
            outputSchemaRef: 'DiagnosticianOutputV1',
            provider: this.config.provider,
            model: this.config.model,
          },
        })
        return
      } catch (err) {
        lastError = err instanceof Error ? err : new Error(String(err))

        if (err instanceof PDRuntimeError) throw err

        if (err instanceof Error && err.name === 'AbortError') {
          this.runs.set(runId, { status: 'failed' })
          throw new PDRuntimeError('timeout', `Run timed out after ${this.config.timeoutMs}ms`)
        }

        if (attempt < maxRetries) {
          // Exponential backoff
          await new Promise(r => setTimeout(r, 1000 * Math.pow(2, attempt)))
          continue
        }
      }
    }

    this.runs.set(runId, { status: 'failed' })
    throw new PDRuntimeError(
      'execution_failed',
      `Failed after ${maxRetries + 1} attempts: ${lastError?.message}`
    )
  }

  async pollRun(runId: string): Promise<RunStatus> {
    const run = this.runs.get(runId)
    if (!run) {
      throw new PDRuntimeError('input_invalid', `Run not found: ${runId}`)
    }
    return run.status
  }

  async cancelRun(runId: string): Promise<void> {
    const run = this.runs.get(runId)
    if (run && run.status === 'running') {
      this.runs.set(runId, { status: 'cancelled' })
    }
  }

  async fetchOutput(runId: string): Promise<StructuredRunOutput> {
    const run = this.runs.get(runId)
    if (!run || !run.output) {
      throw new PDRuntimeError('input_invalid', `Output not available for run: ${runId}`)
    }
    return run.output
  }

  async fetchArtifacts(runId: string): Promise<unknown[]> {
    return []
  }
}
```
