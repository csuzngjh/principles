} catch (error: unknown) {
  const message = error instanceof Error ? error.message : String(error);
  let errorCategory = 'execution_failed';
  if (error instanceof PDRuntimeError) {
    errorCategory = error.category;

      errorCategory,

      runtimeKind,

  process.exit(1);
