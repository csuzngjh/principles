export class MonitoringQueryService {
  constructor(workspaceDir: string);
  dispose(): void;
  getWorkflows(filters: { state?: string; type?: string }): WorkflowListResponse;
  getTrinityStatus(workflowId: string): TrinityStatusResponse | null;
  getTrinityHealth(): TrinityHealthResponse;
}

interface WorkflowListResponse {
  workflows: {
    workflowId: string;
    type: string;
    state: string;
    duration: number;
    createdAt: string;
    stuckDuration: number | null;
  }[];
}

interface TrinityStatusResponse {
  workflowId: string;
  stages: {
    stage: 'dreamer' | 'philosopher' | 'scribe';
    status: 'pending' | 'running' | 'completed' | 'failed';
    outputCount: number;
    duration?: number;
  }[];
}

interface TrinityHealthResponse {
  totalCalls: number;
  avgDuration: number;
  failureRate: number;
  stageStats: {
    dreamer: { total: number; completed: number; failed: number };
    philosopher: { total: number; completed: number; failed: number };
    scribe: { total: number; completed: number; failed: number };
  };
}

if (pathname === `${API_PREFIX}/monitoring/workflows` && method === 'GET') {
  const ms = createMonitoringService(api);
  try {
    const state = url.searchParams.get('state');
    const type = url.searchParams.get('type');
    const result = ms.getWorkflows({
      state: state ?? undefined,
      type: type ?? undefined,
    });
    json(res, 200, result);
    return true;
  } catch (error) {
    api.logger.warn(`[PD:Monitoring] Workflows query failed: ${String(error)}`);
    json(res, 500, { error: 'internal_error', message: String(error) });
    return true;
  } finally {
    ms.dispose();
  }
}
```

if (pathname === `${API_PREFIX}/monitoring/trinity` && method === 'GET') {
  const ms = createMonitoringService(api);
  try {
    const workflowId = url.searchParams.get('workflowId');
    if (!workflowId) {
      json(res, 400, { error: 'bad_request', message: 'workflowId query parameter is required.' });
      return true;
    }
    const result = ms.getTrinityStatus(workflowId);
    if (result === null) {
      json(res, 404, { error: 'not_found', message: `Workflow ${workflowId} not found.` });
      return true;
    }
    json(res, 200, result);
    return true;
  } catch (error) {
    api.logger.warn(`[PD:Monitoring] Trinity status query failed for ${workflowId}: ${String(error)}`);
    json(res, 500, { error: 'internal_error', message: String(error) });
    return true;
  } finally {
    ms.dispose();
  }
}
```

Key implementation details:
- Validate workflowId parameter is present (400 if missing)
- Handle null result from getTrinityStatus() as 404 not found
- Include workflowId in error log for debugging
- Use same try-finally pattern for dispose()
- Return 400 for bad request, 404 for not found, 200 for success, 500 for error
