```typescript
/**
 * KeywordOptimizationService
 *
 * Disptaches LLM subagent via CorrectionObserverWorkflowManager to optimize
 * correction keywords, then applies the returned mutations (add/update/remove)
 * to the CorrectionCueLearner store.
 *
 * Per D-40-01, D-40-02, D-40-03: evolution-worker.ts calls this service,
 * which calls CorrectionCueLearner directly. evolution-worker.ts does NOT
 * call CorrectionCueLearner.
 */

import type { PluginLogger } from '../openclaw-sdk.js';
import type { WorkspaceContext } from './workspace-context.js';
import type { OpenClawPluginApi } from '../openclaw-sdk.js';
import { CorrectionCueLearner } from './correction-cue-learner.js';
import { CorrectionObserverWorkflowManager, correctionObserverWorkflowSpec } from '../service/subagent-workflow/correction-observer-workflow-manager.js';
import type { CorrectionObserverPayload, CorrectionObserverResult } from '../service/subagent-workflow/correction-observer-types.js';
import type { RuntimeDirectDriver } from '../service/subagent-workflow/runtime-direct-driver.js';

const WORKFLOW_TTL_MS = 5 * 60 * 1000; // 5 minutes

export interface KeywordOptimizationState {
  workflowId: string | null;
  status: 'idle' | 'running' | 'completed' | 'failed';
  lastError?: string;
}

/**
 * Build the CorrectionObserverPayload for the LLM subagent.
 * Includes keyword store summary + recent user messages with correctionDetected.
 */
function buildPayload(wctx: WorkspaceContext): CorrectionObserverPayload {
  const learner = CorrectionCueLearner.get(wctx.stateDir);
  const store = learner.getStore();

  // Collect recent user turns with correctionDetected from trajectory
  const recentCorrectionMessages: string[] = [];
  const sessions = wctx.trajectory?.listAllSessions?.() ?? [];
  for (const sessionId of sessions.slice(-10)) { // last 10 sessions
    const turns = wctx.trajectory?.listUserTurnsForSession?.(sessionId) ?? [];
    for (const turn of turns.slice(-5)) { // last 5 turns per session
      if (turn.correctionDetected && turn.rawText) {
        recentCorrectionMessages.push(turn.rawText);
      }
    }
  }

  const keywordStoreSummary = {
    totalKeywords: store.keywords.length,
    terms: store.keywords.map(k => ({
      term: k.term,
      weight: k.weight,
      hitCount: k.hitCount ?? 0,
      truePositiveCount: k.truePositiveCount ?? 0,
      falsePositiveCount: k.falsePositiveCount ?? 0,
    })),
  };

  return {
    parentSessionId: `keyword_optimization:${wctx.stateDir}`,
    workspaceDir: wctx.workspaceDir,
    keywordStoreSummary,
    recentMessages: recentCorrectionMessages.slice(-20), // cap at 20
  };
}

/**
 * Apply mutations from CorrectionObserverResult to the CorrectionCueLearner store.
 * Per D-40-01: KeywordOptimizationService applies mutations.
 * Per Pitfall 4: Do NOT reset TP/FP counters on 'update'.
 */
function applyMutations(learner: CorrectionCueLearner, result: CorrectionObserverResult): number {
  const store = learner.getStore();
  let applied = 0;

  for (const [term, mutation] of Object.entries(result.updates)) {
    const normalizedTerm = term.trim().toLowerCase();
    if (!normalizedTerm || normalizedTerm.length > 100) {
      continue; // Skip invalid terms
    }

    if (mutation.action === 'add') {
      try {
        const weight = mutation.weight !== undefined
          ? Math.max(0.1, Math.min(0.9, mutation.weight)) // clamp per ASVS V5
          : 0.5;
        learner.add({ term: normalizedTerm, weight, source: 'llm' });
        applied++;
      } catch {
        // 200-term limit exceeded — skip
      }
    } else if (mutation.action === 'update') {
      const kw = store.keywords.find(k => k.term.toLowerCase() === normalizedTerm);
      if (kw && mutation.weight !== undefined) {
        // Only update weight — do NOT reset TP/FP counters (Pitfall 4)
        kw.weight = Math.max(0.1, Math.min(0.9, mutation.weight));
        applied++;
      }
    } else if (mutation.action === 'remove') {
      const before = store.keywords.length;
      store.keywords = store.keywords.filter(
        k => k.term.toLowerCase() !== normalizedTerm
      );
      if (store.keywords.length < before) {
        applied++;
      }
    }
  }

  if (applied > 0) {
    learner.flush();
  }
  return applied;
}

/**
 * Run keyword optimization for a workspace.
 *
 * Algorithm:
 * 1. Check throttle (canRunKeywordOptimization)
 * 2. Build payload (keywordStoreSummary + recentMessages)
 * 3. Start workflow via CorrectionObserverWorkflowManager
 * 4. Return immediately (fire-and-poll: caller stores workflowId, polls on next cycle)
 *
 * Returns the workflowId for polling, or null if not started.
 */
export async function runKeywordOptimization(
  wctx: WorkspaceContext,
  api: OpenClawPluginApi | undefined,
  logger: PluginLogger | undefined,
  options: {
    workflowId?: string | null; // existing workflowId to poll, or null to start new
    onComplete?: (applied: number) => void;
    onError?: (error: string) => void;
  } = {}
): Promise<{ workflowId: string | null; status: 'started' | 'polling' | 'completed' | 'throttled' | 'no_api' }> {
  const learner = CorrectionCueLearner.get(wctx.stateDir);

  // Check throttle
  if (!learner.canRunKeywordOptimization()) {
    logger?.debug?.('[PD:KeywordOptimization] Throttled — max 4 optimizations/day');
    return { workflowId: null, status: 'throttled' };
  }

  // Need API for subagent runtime
  if (!api?.runtime?.subagent) {
    logger?.debug?.('[PD:KeywordOptimization] No API or subagent runtime — skipping');
    return { workflowId: null, status: 'no_api' };
  }

  const subagent = api.runtime.subagent as RuntimeDirectDriver['subagent'];
  const agentSession = api.runtime.agent?.session;

  // If we have a workflowId, poll for completion
  if (options.workflowId) {
    const manager = new CorrectionObserverWorkflowManager({
      workspaceDir: wctx.workspaceDir,
      logger: logger!,
      subagent,
      agentSession,
    });

    try {
      const summary = await manager.getWorkflowDebugSummary(options.workflowId);
      if (summary?.state === 'completed') {
        const rawText = summary.latestResult;
        let result: CorrectionObserverResult | null = null;

        // Parse JSON result
        if (rawText?.trim()) {
          try {
            result = JSON.parse(rawText.trim()) as CorrectionObserverResult;
          } catch {
            const match = /\{[\s\S]*\}/.exec(rawText);
            if (match) {
              try {
                result = JSON.parse(match[0]) as CorrectionObserverResult;
              } catch { /* ignore */ }
            }
          }
        }

        if (result) {
          const applied = applyMutations(learner, result);
          learner.recordOptimizationPerformed();
          logger?.info?.(`[PD:KeywordOptimization] Applied ${applied} mutations: ${result.summary}`);
          options.onComplete?.(applied);
          return { workflowId: null, status: 'completed' };
        }
      } else if (summary?.state === 'terminal_error') {
        logger?.warn?.(`[PD:KeywordOptimization] Workflow failed: ${summary.error}`);
        options.onError?.(summary.error ?? 'unknown error');
        return { workflowId: null, status: 'completed' };
      }
      // Otherwise still running — return workflowId to keep polling
      return { workflowId: options.workflowId, status: 'polling' };
    } finally {
      manager.dispose();
    }
  }

  // Start new workflow
  const payload = buildPayload(wctx);

  const manager = new CorrectionObserverWorkflowManager({
    workspaceDir: wctx.workspaceDir,
    logger: logger!,
    subagent,
    agentSession,
  });

  try {
    // Surface degrade: skip boot sessions (same check as CorrectionObserverWorkflowManager)
    if (payload.parentSessionId.startsWith('boot-')) {
      logger?.debug?.('[PD:KeywordOptimization] Skipping: boot session');
      return { workflowId: null, status: 'no_api' };
    }

    const handle = await manager.startWorkflow(correctionObserverWorkflowSpec, {
      parentSessionId: payload.parentSessionId,
      workspaceDir: wctx.workspaceDir,
      taskInput: payload,
    });

    logger?.info?.(`[PD:KeywordOptimization] Started workflow ${handle.workflowId}`);
    return { workflowId: handle.workflowId, status: 'started' };
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err);
    logger?.warn?.(`[PD:KeywordOptimization] Failed to start workflow: ${msg}`);
    options.onError?.(msg);
    return { workflowId: null, status: 'failed' };
  } finally {
    manager.dispose();
  }
}
```
