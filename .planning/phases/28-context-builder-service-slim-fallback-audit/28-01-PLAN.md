    async buildCycleContext(
      wctx: WorkspaceContext,
      logger?: PluginLogger,
      eventLog?: EventLog,
    ): Promise<CycleContextResult> {
      const errors: string[] = [];

      // Permissive validation
      if (!wctx || typeof wctx !== 'object') {
        errors.push('Invalid workspace context');
        return {
          idle: { isIdle: false, mostRecentActivityAt: 0, idleForMs: 0, userActiveSessions: 0, abandonedSessionIds: [], trajectoryGuardrailConfirmsIdle: false, reason: 'invalid_wctx' },
          cooldown: { globalCooldownActive: false, globalCooldownUntil: null, globalCooldownRemainingMs: 0, principleCooldownActive: false, principleCooldownUntil: null, principleCooldownRemainingMs: 0, quotaExhausted: false, runsRemaining: 0 },
          recentPain: { mostRecent: null, recentPainCount: 0, recentMaxPainScore: 0 },
          activeSessions: [],
          errors,
        };
      }

      // checkWorkspaceIdle — wrapped in try/catch
      // FB-04: checkWorkspaceIdle failure is FAIL-VISIBLE (CONTRACT-05) — emit skip event, return default, pipeline continues
      let idle: IdleCheckResult;
      try {
        idle = checkWorkspaceIdle(wctx.workspaceDir, {});
      } catch (err) {
        errors.push(`checkWorkspaceIdle failed: ${String(err)}`);
        idle = { isIdle: false, mostRecentActivityAt: 0, idleForMs: 0, userActiveSessions: 0, abandonedSessionIds: [], trajectoryGuardrailConfirmsIdle: false, reason: 'checkWorkspaceIdle_error' };
        // FAIL-VISIBLE: emit skip event so diagnostics can observe this degradation
        if (eventLog) {
          eventLog.recordSkip(undefined, {
            reason: 'checkWorkspaceIdle_error',
            fallback: 'default_idle_assumption',
            context: { error: String(err) },
          });
        }
      }

      // checkCooldown — wrapped in try/catch
      // FB-05: checkCooldown failure is FAIL-VISIBLE (CONTRACT-05) — emit skip event, return default, pipeline continues
      let cooldown: CooldownCheckResult;
      try {
        cooldown = checkCooldown(wctx.stateDir);
      } catch (err) {
        errors.push(`checkCooldown failed: ${String(err)}`);
        cooldown = { globalCooldownActive: false, globalCooldownUntil: null, globalCooldownRemainingMs: 0, principleCooldownActive: false, principleCooldownUntil: null, principleCooldownRemainingMs: 0, quotaExhausted: false, runsRemaining: 0 };
        // FAIL-VISIBLE: emit skip event so diagnostics can observe this degradation
        if (eventLog) {
          eventLog.recordSkip(undefined, {
            reason: 'checkCooldown_error',
            fallback: 'no_cooldown_assumption',
            context: { error: String(err) },
          });
        }

      // extractRecentPainContext — wrapped in try/catch
      let recentPain: RecentPainContext;
      try {
        recentPain = new PainFlagDetector(this.workspaceDir).extractRecentPainContext();

        });
      } catch (err) {

      }

    }
