```typescript
// RuleHostInput: frozen snapshot provided to implementations (D-05, D-06, Pattern 4)
export interface RuleHostInput {
  action: {
    toolName: string;
    normalizedPath: string | null;
    paramsSummary: Record<string, unknown>;
  };
  workspace: {
    isRiskPath: boolean;
    planStatus: 'NONE' | 'DRAFT' | 'READY' | 'UNKNOWN';
    hasPlanFile: boolean;
  };
  session: {
    sessionId?: string;
    currentGfi: number;
    recentThinking: boolean;
  };
  evolution: {
    epTier: number;
  };
  derived: {
    estimatedLineChanges: number;
    bashRisk: 'safe' | 'normal' | 'dangerous' | 'unknown';
  };
}

// RuleHostDecision: limited to three outcomes per D-07
export type RuleHostDecision = 'allow' | 'block' | 'requireApproval';

// RuleHostMeta: exported by each implementation for identification
export interface RuleHostMeta {
  name: string;
  version: string;
  ruleId: string;
  coversCondition: string;
}

// RuleHostResult: structured output from a single implementation evaluation
export interface RuleHostResult {
  decision: RuleHostDecision;
  matched: boolean;
  reason: string;
  diagnostics?: Record<string, unknown>;
}

// LoadedImplementation: a successfully loaded active implementation
export interface LoadedImplementation {
  implId: string;
  ruleId: string;
  meta: RuleHostMeta;
  evaluate: (input: RuleHostInput) => RuleHostResult;
}
```

**2. Create `packages/openclaw-plugin/src/core/rule-host-helpers.ts`**

Implement the minimal helper whitelist per D-15 and D-16. The helpers receive pre-computed values, NOT live workspace handles.

```typescript
import type { RuleHostInput } from './rule-host-types.js';

export interface RuleHostHelpers {
  isRiskPath(): boolean;
  getToolName(): string;
  getEstimatedLineChanges(): number;
  getBashRisk(): 'safe' | 'normal' | 'dangerous' | 'unknown';
  hasPlanFile(): boolean;
  getPlanStatus(): 'NONE' | 'DRAFT' | 'READY' | 'UNKNOWN';
  getCurrentEpiTier(): number;
}

export function createRuleHostHelpers(input: RuleHostInput): RuleHostHelpers {
  return Object.freeze({
    isRiskPath: () => input.workspace.isRiskPath,
    getToolName: () => input.action.toolName,
    getEstimatedLineChanges: () => input.derived.estimatedLineChanges,
    getBashRisk: () => input.derived.bashRisk,
    hasPlanFile: () => input.workspace.hasPlanFile,
    getPlanStatus: () => input.workspace.planStatus,
    getCurrentEpiTier: () => input.evolution.epTier,
  });
}
```

**3. Create `packages/openclaw-plugin/src/core/rule-host.ts`**

Implement the RuleHost class. This class:
- Constructor takes `stateDir: string`
- `evaluate(input: RuleHostInput): RuleHostResult | undefined` method
- On evaluate: loads active code implementations from ledger (findActiveImplementation / listImplementationsByLifecycleState filtered to type=code and lifecycleState=active)
- For each active implementation: if the implementation has a code asset path, attempt to load and execute it via `node:vm`
- Loading uses `vm.compileFunction` with `codeGeneration: { strings: false, wasm: false }`, no dynamic import support
- The vm context receives only `{ helpers: frozenHelpers }` — no fs, process, require, global, etc.
- Decision merge: iterate implementations, if any returns `block` then short-circuit and return block, if any returns `requireApproval` collect it, if all return `allow` or `matched: false` then return undefined (no opinion)
- On ANY host error (load failure, eval error, vm error): log the error and return undefined (conservative degradation per D-08). Never throw, never bypass downstream gates.
- Do NOT implement replay, promotion, or nocturnal artifact generation (D-12, D-13, D-14).

For now, since actual code assets do not yet exist on disk (Phase 14 will create them), the host must handle the case where no active code implementations exist gracefully by returning undefined. The host should also handle the case where an implementation's asset path does not exist or is unreadable by catching the error and continuing.
