## Goal

### Step 2: Add `statsField` dot-path resolver

**File:** `packages/openclaw-plugin/src/service/runtime-summary-service.ts`

```typescript
function resolveStatsField(stats: DailyStats, dotPath: string): number {
  const parts = dotPath.split('.');
  let current: unknown = stats;
  for (const part of parts) {
    if (current == null || typeof current !== 'object') return 0;
    current = (current as Record<string, unknown>)[part];
  }
  return typeof current === 'number' ? current : 0;
}

function buildWarnings(dotPath: string): string[] {
  return [`statsField not resolvable: ${dotPath}`];
}
```

### Step 3: Update `getSummary()` signature and logic

```typescript
interface GetSummaryOptions {
  loaderWarnings?: string[];   // existing
  funnels?: Map<string, WorkflowStage[]>;  // NEW
}

getSummary(options: GetSummaryOptions = {}): RuntimeSummary {
  const { loaderWarnings = [], funnels } = options;

  // Load dailyStats from fs (existing logic)...

  // Build workflowFunnels if funnels provided
  let workflowFunnels: WorkflowFunnelOutput[] | undefined;
  if (funnels) {
    workflowFunnels = [];
    for (const [funnelKey, stages] of funnels.entries()) {
      const funnelLabel = this.getFunnelLabel(funnelKey);
      const stageOutputs: WorkflowFunnelStageOutput[] = stages.map(stage => {
        const count = resolveStatsField(dailyStats, stage.statsField);
        return {
          key: stage.key,
          label: stage.label,
          statsField: stage.statsField,
          count,
        };
      });
      workflowFunnels.push({ funnelKey, funnelLabel, stages: stageOutputs });
    }
  }

  // Build warnings
  const warnings = [...loaderWarnings];
  if (workflowFunnels) {
    for (const funnel of workflowFunnels) {
      for (const stage of funnel.stages) {
        if (stage.count === 0) {
          warnings.push(`statsField not resolvable: ${stage.statsField}`);
        }
      }
    }
  }

  const status = warnings.length > 0 ? 'degraded' : 'ok';

  return {
    // ... existing stats fields ...
    ...(workflowFunnels && { workflowFunnels }),
    metadata: { generatedAt: new Date().toISOString(), status, warnings },
  };
}

private getFunnelLabel(funnelKey: string): string {
  // Human-readable label — could come from YAML funnel metadata if added later
  return funnelKey;  // For now, use key as label
}
```
