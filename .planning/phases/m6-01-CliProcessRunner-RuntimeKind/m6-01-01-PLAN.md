import { spawn, type SpawnOptions } from 'child_process';

export interface CliProcessRunnerOptions {
  command: string;
  args?: string[];
  cwd?: string;
  env?: Record<string, string>;
  timeoutMs?: number;
  killGracePeriodMs?: number;
}

export interface CliOutput {
  stdout: string;
  stderr: string;
  exitCode: number | null;
  timedOut: boolean;
  durationMs: number;
}

export async function runCliProcess(opts: CliProcessRunnerOptions): Promise<CliOutput> {
  // Implementation must:
  // 1. Validate: command must be non-empty string (throw TypeError if empty/invalid)
  // 2. Use spawn(command, args ?? [], { cwd, env: merged, shell: false, detached: true })
  // 3. Merge env: Object.assign({}, process.env, opts.env ?? {})
  // 4. Set detached: true — needed for process group semantics (kill -pid works on group)
  // 5. Attach stdout/stderr data handlers BEFORE spawn
  // 6. Track startedAt = Date.now() for durationMs
  // 7. If timeoutMs != null: setTimeout kills process tree on expiry
  //    - Windows: spawn('taskkill', ['/PID', String(pid), '/T', '/F'])
  //    - Unix: process.kill(-pid, 'SIGTERM') then setTimeout killGracePeriodMs → SIGKILL
  // 8. Resolve on proc 'close' event with { stdout, stderr, exitCode, timedOut, durationMs }
  // 9. Handle proc 'error' event (ENOENT etc.) — resolve with exitCode: null and empty stdout
  // 10. return CliOutput (stdout as returned string, stderr as internal capture string)
}
```

Implementation guidance from RESEARCH.md:
- Use cross-platform kill: `process.platform === 'win32'` check
- For Windows tree kill: `spawn('taskkill', ['/PID', String(pid), '/T', '/F'])`
- For Unix tree kill: `process.kill(-pid, 'SIGTERM')` then grace period then `kill(-pid, 'SIGKILL')`
- detached: true is REQUIRED for process group semantics (A1 from RESEARCH.md)
- Always clear timeout on normal exit to avoid orphaned timers

import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import { runCliProcess, type CliOutput } from './cli-process-runner.js';

describe('runCliProcess', () => {
  // Use fake timers for timeout tests
  beforeEach(() => { vi.useFakeTimers(); });
  afterEach(() => { vi.restoreAllMocks(); });

  describe('success case (exit code 0)', () => {
    it('resolves with stdout, exitCode 0, timedOut false, durationMs > 0', async () => {
      const result = await runCliProcess({
        command: 'node',
        args: ['-e', 'console.log("hello")'],
      });
      expect(result.stdout).toBe('hello\n');
      expect(result.exitCode).toBe(0);
      expect(result.timedOut).toBe(false);
      expect(result.durationMs).toBeGreaterThanOrEqual(0);
    });
  });

  describe('non-zero exit case', () => {
    it('captures non-zero exit code in result', async () => {
      const result = await runCliProcess({
        command: 'node',
        args: ['-e', 'process.exit(42)'],
      });
      expect(result.exitCode).toBe(42);
      expect(result.timedOut).toBe(false);
    });
  });

  describe('timeout case', () => {
    it('sets timedOut=true and kills the process', async () => {
      const runPromise = runCliProcess({
        command: 'node',
        args: ['-e', 'setTimeout(() => {}, 10000)'],
        timeoutMs: 100,
        killGracePeriodMs: 50,
      });

      // Advance time past timeout
      await vi.advanceTimersByTime(200);
      const result = await runPromise;

      expect(result.timedOut).toBe(true);
      expect(result.exitCode).toBeNull(); // killed, not exited
    });
  });

  describe('ENOENT case (binary not found)', () => {
    it('resolves with exitCode null and empty stdout on ENOENT', async () => {
      const result = await runCliProcess({
        command: 'nonexistent-binary-xyz123',
      });
      expect(result.exitCode).toBeNull();
      expect(result.stdout).toBe('');
    });
  });
});
```
