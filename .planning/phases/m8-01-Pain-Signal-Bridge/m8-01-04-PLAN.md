Create a new file: packages/principles-core/src/runtime-v2/pain-signal-bridge.ts

This file implements the pain signal bridge — a service that:
1. Receives a PainSignal from the plugin hook (pain.ts)
2. Creates a task in SqliteTaskStore with taskKind='diagnostician'
3. Creates a run in SqliteRunStore
4. Invokes DiagnosticianRunner.run(taskId)
5. On success: triggers CandidateIntakeService.intake() for each principle candidate (happy path)

```typescript
import type { PainSignal } from '../../openclaw-plugin/src/core/pain.js';
import type { DiagnosticianRunner } from './runner/diagnostician-runner.js';
import type { DiagnosticianCommitter } from './store/diagnostician-committer.js';
import type { CandidateIntakeService } from './candidate-intake-service.js';
import type { RuntimeStateManager } from './store/runtime-state-manager.js';
import type { TelemetryEvent } from '../../telemetry-event.js';

export interface PainSignalBridgeOptions {
  stateManager: RuntimeStateManager;
  runner: DiagnosticianRunner;
  committer: DiagnosticianCommitter;
  intakeService: CandidateIntakeService;
  /** If true, intake is triggered automatically after successful commit (happy path). Default: true */
  autoIntakeEnabled?: boolean;
  /** Emit telemetry events */
  emitEvent?: (event: TelemetryEvent) => void;
}

export class PainSignalBridge {
  readonly #opts: PainSignalBridgeOptions;
  readonly #autoIntakeEnabled: boolean;
  readonly #emitEvent?: (event: TelemetryEvent) => void;

  constructor(opts: PainSignalBridgeOptions) {
    this.#opts = opts;
    this.#autoIntakeEnabled = opts.autoIntakeEnabled ?? true;
    this.#emitEvent = opts.emitEvent;
  }

  /**
   * Process a pain signal: create task, run diagnostician, commit, intake.
   *
   * @param painSignal - The pain signal to process
   * @param workspaceDir - The workspace directory for state storage
   * @returns The outcome including runner result and intake results
   */
  async processPainSignal(
    painSignal: PainSignal,
    workspaceDir: string,
  ): Promise<{
    taskId: string;
    runId: string;
    runnerResult: Awaited<ReturnType<DiagnosticianRunner['run']>>;
    intakeResults: { candidateId: string; ledgerEntry?: unknown }[];
  }> {
    const taskId = `diag_${painSignal.painId}`;

    // Emit: pain_signal_bridged
    this.#emitEvent?.({
      eventType: 'pain_signal_bridged',
      traceId: taskId,
      timestamp: new Date().toISOString(),
      sessionId: painSignal.source,
      agentId: 'pain_signal_bridge',
      payload: {
        painId: painSignal.painId,
        source: painSignal.source,
        score: painSignal.score,
        taskId,
      },
    });

    // 1. Create task in SqliteTaskStore (pain_diagnosis kind)
    const task = await this.#opts.stateManager.createTask({
      taskKind: 'diagnostician',
      status: 'pending',
      input: {
        painSignal,
        painId: painSignal.painId,
        source: painSignal.source,
        reason: painSignal.reason,
        score: painSignal.score,
        severity: painSignal.severity ?? 'moderate',
        trigger: 'pain_signal_bridge',
      },
      owner: 'pain_signal_bridge',
    });

    // 2. Create run in SqliteRunStore
    const run = await this.#opts.stateManager.createRun({
      taskId: task.taskId,
      status: 'in_progress',
    });

    // 3. Invoke DiagnosticianRunner.run(taskId)
    const runnerResult = await this.#opts.runner.run(task.taskId);

    // 4. If succeeded and autoIntakeEnabled, trigger intake for each principle candidate
    const intakeResults: { candidateId: string; ledgerEntry?: unknown }[] = [];
    if (runnerResult.status === 'succeeded' && this.#autoIntakeEnabled) {
      const principleRecs = runnerResult.output.recommendations?.filter(r => r.kind === 'principle') ?? [];
      for (const rec of principleRecs) {
        // Candidate ID format: cand_{commitId}_{index}
        const candidateId = `cand_${runnerResult.contextHash}_${intakeResults.length}`;
        try {
          const ledgerEntry = await this.#opts.intakeService.intake(candidateId);
          intakeResults.push({ candidateId, ledgerEntry });
        } catch (intakeError) {
          // Log but don't fail — intake is happy path, not blocker
          this.#emitEvent?.({
            eventType: 'intake_failed',
            traceId: taskId,
            timestamp: new Date().toISOString(),
            sessionId: painSignal.source,
            agentId: 'pain_signal_bridge',
            payload: {
              candidateId,
              error: intakeError instanceof Error ? intakeError.message : String(intakeError),
            },
          });
          intakeResults.push({ candidateId, ledgerEntry: undefined });
        }
      }
    }

    return {
      taskId: task.taskId,
      runId: run.runId,
      runnerResult,
      intakeResults,
    };
  }
}
```
