**CRITICAL ARCHITECTURE DECISIONS:**

1. **Self-contained input type — NO openclaw-plugin import.**
   principles-core must NOT import PainFlagData from openclaw-plugin (cross-package coupling).
   Define BridgePainSignalInput as a self-contained interface:

```typescript
export interface BridgePainSignalInput {
  source: string;
  score: string | number;  // PainFlagData.score is string; normalize to number
  time: string;              // ISO 8601
  reason: string;
  session_id?: string;
  sessionId?: string;        // Some callers use sessionId
  agent_id?: string;
  trace_id?: string;
  pain_event_id?: string;
}
```

2. **Runner manages run lifecycle — DO NOT call createRun.**
   DiagnosticianRunner.run() internally calls stateManager.acquireLease() which creates the RunRecord.
   Bridge creates task → runner.run(taskId) → runner creates run internally.
   Calling createRun in the bridge would create a duplicate orphan run.

3. **Real candidateId discovery via stateManager.getCandidatesByTaskId().**
   After runner succeeds, query stateManager.getCandidatesByTaskId(taskId) to get CandidateRecord[].
   CandidateRecord.candidateId is a real UUID from randomUUID() in committer — NOT synthetic.
   Call intakeService.intake(realCandidateId) for each.

4. **autoIntakeEnabled (default: true) for happy path (HG-2).**
   When false, skip intake (debug mode) but still complete the runner invocation.

```typescript
import type { DiagnosticianRunner } from './runner/diagnostician-runner.js';
import type { CandidateIntakeService } from './candidate-intake-service.js';
import type { RuntimeStateManager } from './store/runtime-state-manager.js';
import type { TelemetryEvent } from '../../telemetry-event.js';

export { PainSignalBridge } from './pain-signal-bridge.js';

export interface PainSignalBridgeOptions {
  stateManager: RuntimeStateManager;
  runner: DiagnosticianRunner;
  /** Runner owns commit — bridge does NOT depend on committer directly */
  intakeService: CandidateIntakeService;
  /** If true, intake is triggered automatically after successful commit (happy path, HG-2). Default: true */
  autoIntakeEnabled?: boolean;
  emitEvent?: (event: TelemetryEvent) => void;
}

/**
 * Input type for pain signal bridge.
 * Self-contained — does NOT import PainFlagData from openclaw-plugin.
 * Maps from PainFlagData fields which use snake_case and score as string.
 */
export interface BridgePainSignalInput {
  source: string;
  /** Score from PainFlagData — string (e.g., "100") or already numeric */
  score: string | number;
  /** ISO 8601 timestamp */
  time: string;
  reason: string;
  session_id?: string;
  sessionId?: string;
  agent_id?: string;
  trace_id?: string;
  pain_event_id?: string;
}

export class PainSignalBridge {
  readonly #opts: PainSignalBridgeOptions;
  readonly #autoIntakeEnabled: boolean;
  readonly #emitEvent?: (event: TelemetryEvent) => void;

  constructor(opts: PainSignalBridgeOptions) {
    this.#opts = opts;
    this.#autoIntakeEnabled = opts.autoIntakeEnabled ?? true;
    this.#emitEvent = opts.emitEvent;
  }

  /**
   * Process a pain signal: create task, invoke runner, query real candidates, trigger intake.
   *
   * @param input - Pain signal input (from PainFlagData mapping)
   * @param workspaceDir - Workspace directory for state storage
   * @returns Task ID, runner result, and intake results
   */
  async processPainSignal(
    input: BridgePainSignalInput,
    workspaceDir: string,
  ): Promise<{
    taskId: string;
    runnerResult: Awaited<ReturnType<DiagnosticianRunner['run']>>;
    intakeResults: { candidateId: string; ledgerEntry?: unknown }[];
  }> {
    const painId = input.pain_event_id ?? `pain_${Date.now()}`;
    const taskId = `diag_${painId}`;

    // Emit telemetry event
    this.#emitEvent?.({
      eventType: 'pain_signal_bridged',
      traceId: taskId,
      timestamp: new Date().toISOString(),
      sessionId: input.session_id ?? input.sessionId ?? '',
      agentId: 'pain_signal_bridge',
      payload: { painId, source: input.source, score: String(input.score), taskId },
    });

    // 1. Create pending task in SqliteTaskStore (diagnostician kind)
    // NOTE: Do NOT call stateManager.createRun() — runner.run() calls acquireLease() which creates the run internally
    const task = await this.#opts.stateManager.createTask({
      taskKind: 'diagnostician',
      status: 'pending',
      input: {
        source: input.source,
        score: typeof input.score === 'string' ? parseInt(input.score, 10) || 0 : input.score,
        time: input.time,
        reason: input.reason,
        session_id: input.session_id ?? input.sessionId ?? '',
        agent_id: input.agent_id,
        trace_id: input.trace_id,
        pain_event_id: input.pain_event_id,
        trigger: 'pain_signal_bridge',
      },
      owner: 'pain_signal_bridge',
    });

    // 2. Invoke DiagnosticianRunner.run(taskId)
    // Runner internally calls stateManager.acquireLease() which creates the RunRecord — do NOT call createRun
    const runnerResult = await this.#opts.runner.run(task.taskId);

    // 3. Query real candidateIds from state store after runner commits
    // Intake is per-candidate idempotent — partial intake is acceptable.
    // If candidate 1 intake succeeds and candidate 2 fails, candidate 1's ledger
    // entry remains. Failed intakes leave the candidate pending/retryable.
    // intake_failed telemetry is emitted for each failed candidate.
    const intakeResults: { candidateId: string; ledgerEntry?: unknown }[] = [];
    if (runnerResult.status === 'succeeded' && this.#autoIntakeEnabled) {
      const candidates = await this.#opts.stateManager.getCandidatesByTaskId(task.taskId);
      for (const candidate of candidates) {
        try {
          // candidate.candidateId is a real UUID from committer.randomUUID() — NOT synthetic
          const ledgerEntry = await this.#opts.intakeService.intake(candidate.candidateId);
          intakeResults.push({ candidateId: candidate.candidateId, ledgerEntry });
        } catch (intakeError) {
          // Log but don't fail — intake is happy path, not blocker
          // Emit intake_failed telemetry with candidateId for monitoring
          this.#emitEvent?.({
            eventType: 'intake_failed',
            traceId: taskId,
            timestamp: new Date().toISOString(),
            sessionId: input.session_id ?? input.sessionId ?? '',
            agentId: 'pain_signal_bridge',
            payload: {
              candidateId: candidate.candidateId,
              error: intakeError instanceof Error ? intakeError.message : String(intakeError),
            },
          });
          intakeResults.push({ candidateId: candidate.candidateId, ledgerEntry: undefined });
        }
      }
    }

    return { taskId: task.taskId, runnerResult, intakeResults };
  }
}
```

```typescript
wctx.evolutionReducer.on('pain_detected', async (evt) => {
  if (evt.type !== 'pain_detected') return;

  // Map PainFlagData fields to BridgePainSignalInput

    source: painFlag.source ?? '',
    score: painFlag.score ?? '0',
```
