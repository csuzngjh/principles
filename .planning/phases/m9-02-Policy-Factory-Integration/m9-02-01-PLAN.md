## Step 2: Update PainSignalRuntimeFactory

Modify `createPainSignalBridge()` to read runtimeKind from policy:

```typescript
import { PiAiRuntimeAdapter } from './adapter/pi-ai-runtime-adapter.js'
import { OpenClawCliRuntimeAdapter } from './adapter/openclaw-cli-runtime-adapter.js'
import { WorkflowFunnelLoader } from '../workflow-funnel-loader.js'

export function createPainSignalBridge(stateDir: string, workspace: string) {
  const loader = new WorkflowFunnelLoader(stateDir)
  const funnel = loader.loadFunnel('pd-runtime-v2-diagnosis')
  const policy = funnel?.policy || {}

  const runtimeKind = policy.runtimeKind || 'openclaw-cli'

  let adapter: PDRuntimeAdapter
  switch (runtimeKind) {
    case 'pi-ai':
      adapter = new PiAiRuntimeAdapter({
        provider: policy.provider || 'openrouter',
        model: policy.model || 'anthropic/claude-sonnet-4',
        apiKeyEnv: policy.apiKeyEnv || 'OPENROUTER_API_KEY',
        maxRetries: policy.maxRetries ?? 2,
        timeoutMs: policy.timeoutMs ?? 120_000,
        workspace,
      })
      break
    case 'test-double':
      // For testing
      adapter = new TestDoubleRuntimeAdapter()
      break
    case 'openclaw-cli':
    default:
      adapter = new OpenClawCliRuntimeAdapter({
        runtimeMode: 'local',
        workspace,
        timeoutMs: policy.timeoutMs ?? 240_000,
      })
      break
  }

  // ... rest of bridge creation
}
```
