    3. **`assemble(taskId)` method implementation — assembly flow:**

       a. Get task from TaskStore:
          ```typescript
          const task = await this.taskStore.getTask(taskId);
          if (!task) {
            throw new PDRuntimeError('storage_unavailable', `Task not found: ${taskId}`);
          }
          ```

       b. Validate taskKind:
          ```typescript
          if (task.taskKind !== 'diagnostician') {
            throw new PDRuntimeError(
              'input_invalid',
              `Task ${taskId} is not a diagnostician task (kind: ${task.taskKind})`,
            );
          }
          const dt = task as DiagnosticianTaskRecord;
          ```

       c. Query history:
          ```typescript
          const historyResult = await this.historyQuery.query(taskId);
          ```

       d. Get runIds for sourceRefs:
          ```typescript
          const runs = await this.runStore.listRunsByTask(taskId);
          const runIds = runs.map(r => r.runId);
          ```

       e. Generate contextId and contextHash:
          ```typescript
          const contextId = randomUUID();
          const serialized = JSON.stringify(historyResult.entries);
          const contextHash = createHash('sha256').update(serialized).digest('hex');
          ```

       f. Build diagnosisTarget from DiagnosticianTaskRecord:
          ```typescript
          const diagnosisTarget: DiagnosisTarget = {
            reasonSummary: dt.reasonSummary || undefined,
            source: dt.source || undefined,
            severity: dt.severity || undefined,
            painId: dt.sourcePainId || undefined,
            sessionIdHint: dt.sessionIdHint || undefined,
          };
          ```

       g. Generate ambiguityNotes:
          ```typescript
          const ambiguityNotes = this.buildAmbiguityNotes(taskId, historyResult.entries, historyResult.truncated);
          ```

       h. Create private helper `buildAmbiguityNotes`:
          - Check if conversationWindow is empty → add "No conversation history available for diagnostician task {taskId}"
          - Check if truncated → add "Conversation window truncated; some history may be missing"
          - Count entries with undefined text → add "N entries have empty text content"
          - Return undefined if no notes generated (not empty array)

       i. Assemble final payload:
          ```typescript
          const payload: DiagnosticianContextPayload = {
            contextId,
            contextHash,
            taskId,
            workspaceDir: dt.workspaceDir,
            sourceRefs: [taskId, ...runIds],
            diagnosisTarget,
            conversationWindow: historyResult.entries,
            ambiguityNotes,
          };
          ```

       j. Validate with TypeBox:
          ```typescript
          if (!Value.Check(DiagnosticianContextPayloadSchema, payload)) {
            throw new PDRuntimeError('storage_unavailable', 'Context payload schema validation failed');
          }
          return payload;
          ```
  </action>
