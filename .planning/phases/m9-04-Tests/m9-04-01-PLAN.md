Create `packages/principles-core/src/runtime-v2/adapter/__tests__/pi-ai-runtime-adapter.test.ts`:

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { PiAiRuntimeAdapter } from '../pi-ai-runtime-adapter.js'

// Mock @mariozechner/pi-ai
vi.mock('@mariozechner/pi-ai', () => ({
  getModel: vi.fn(),
  complete: vi.fn(),
}))

describe('PiAiRuntimeAdapter', () => {
  const config = {
    provider: 'openrouter',
    model: 'anthropic/claude-sonnet-4',
    apiKeyEnv: 'TEST_API_KEY',
    maxRetries: 2,
    timeoutMs: 5000,
    workspace: '/tmp/test',
  }

  beforeEach(() => {
    process.env.TEST_API_KEY = 'test-key'
  })

  it('kind() returns pi-ai', () => {
    const adapter = new PiAiRuntimeAdapter(config)
    expect(adapter.kind()).toBe('pi-ai')
  })

  it('healthCheck returns healthy when API key exists', async () => {
    const adapter = new PiAiRuntimeAdapter(config)
    const health = await adapter.healthCheck()
    expect(health.healthy).toBe(true)
  })

  it('healthCheck returns unhealthy when API key missing', async () => {
    delete process.env.TEST_API_KEY
    const adapter = new PiAiRuntimeAdapter(config)
    const health = await adapter.healthCheck()
    expect(health.healthy).toBe(false)
  })

  describe('startRun', () => {
    it('throws runtime_unavailable when API key missing', async () => {
      delete process.env.TEST_API_KEY
      const adapter = new PiAiRuntimeAdapter(config)
      await expect(adapter.startRun({ prompt: 'test' })).rejects.toThrow('runtime_unavailable')
    })

    it('succeeds with valid DiagnosticianOutputV1 response', async () => {
      // Mock complete to return valid output
      const { complete } = await import('@mariozechner/pi-ai')
      vi.mocked(complete).mockResolvedValue({
        content: [{ type: 'text', text: JSON.stringify({
          valid: true,
          diagnosisId: 'diag-1',
          taskId: 'task-1',
          summary: 'Test diagnosis',
          rootCause: 'test',
          violatedPrinciples: [],
          evidence: [],
          recommendations: [],
          confidence: 0.9,
        })}],
        usage: { promptTokens: 100, completionTokens: 50 },
        stopReason: 'end_turn',
      })

      const adapter = new PiAiRuntimeAdapter(config)
      const handle = await adapter.startRun({
        prompt: 'test prompt',
        systemPrompt: 'test system',
      })
      expect(handle.status).toBe('running')
      expect(handle.runId).toBeTruthy()
    })

    it('throws execution_failed after maxRetries exhaustion', async () => {
      const { complete } = await import('@mariozechner/pi-ai')
      vi.mocked(complete).mockRejectedValue(new Error('API error'))

      const adapter = new PiAiRuntimeAdapter({ ...config, maxRetries: 0 })
      const handle = await adapter.startRun({ prompt: 'test' })

      // Wait for async execution
      await new Promise(r => setTimeout(r, 100))

      const status = await adapter.pollRun(handle.runId)
      expect(status).toBe('failed')
    })

    it('throws timeout when AbortSignal fires', async () => {
      const { complete } = await import('@mariozechner/pi-ai')
      vi.mocked(complete).mockImplementation(() =>
        new Promise((_, reject) => setTimeout(() => reject(new Error('Aborted')), 100))
      )

      const adapter = new PiAiRuntimeAdapter({ ...config, timeoutMs: 50 })
      const handle = await adapter.startRun({ prompt: 'test' })

      await new Promise(r => setTimeout(r, 200))
      const status = await adapter.pollRun(handle.runId)
      expect(status).toBe('failed')
    })

    it('throws output_invalid for non-JSON response', async () => {
      const { complete } = await import('@mariozechner/pi-ai')
      vi.mocked(complete).mockResolvedValue({
        content: [{ type: 'text', text: 'not json' }],
        usage: { promptTokens: 100, completionTokens: 50 },
        stopReason: 'end_turn',
      })

      const adapter = new PiAiRuntimeAdapter({ ...config, maxRetries: 0 })
      const handle = await adapter.startRun({ prompt: 'test' })

      await new Promise(r => setTimeout(r, 100))
      const status = await adapter.pollRun(handle.runId)
      expect(status).toBe('failed')
    })
  })

  describe('pollRun', () => {
    it('returns current status for known run', async () => {
      const adapter = new PiAiRuntimeAdapter(config)
      // ... test poll behavior
    })

    it('throws input_invalid for unknown runId', async () => {
      const adapter = new PiAiRuntimeAdapter(config)
      await expect(adapter.pollRun('unknown')).rejects.toThrow('input_invalid')
    })
  })

  describe('fetchArtifacts', () => {
    it('returns empty array', async () => {
      const adapter = new PiAiRuntimeAdapter(config)
      const artifacts = await adapter.fetchArtifacts('any-run')
      expect(artifacts).toEqual([])
    })
  })
})
```
